```powershell
$config = New-PesterConfiguration

$config.TestResult.Enabled = $true
$config.TestResult.OutputFormat = 'NUnitXml'
$config.TestResult.OutputPath = './Tests/Artifacts/test-results.xml'

# Invoke with configuration
Invoke-Pester -Configuration $config
```

```powershell
BeforeAll {

Describe 'Test-MDEServiceStatus' {

                    StartType = 'Automatic'

            }

        }
    }
}
```

## Step 2: Create Tests/Mapping/generate-mapping.ps1
This script generates and validates function-to-test mapping for TEST-01 verification.

Contents:
```powershell
#Requires -Version 5.1

[CmdletBinding()]
param()

# Get all public functions from MDEValidator/Public folder
\$publicFunctions = Get-ChildItem './MDEValidator/Public/*.ps1' -ErrorAction SilentlyContinue | 
    Select-Object -ExpandProperty BaseName | 
    Sort-Object

# Get all test files from Tests/Public folder
\$testFiles = Get-ChildItem './Tests/Public/*.Tests.ps1' -ErrorAction SilentlyContinue | 
    Select-Object -ExpandProperty BaseName | 
    ForEach-Object { \$_ -replace '\.Tests$', '' } | 
    Sort-Object

# Build mapping
\$mapping = [ordered]@{
    generated_at = Get-Date -Format 'o'
    public_function_count = \$publicFunctions.Count
    test_file_count = \$testFiles.Count
    functions = [ordered]@{}
}

foreach (\$func in \$publicFunctions) {
    \$testExists = \$func -in \$testFiles
    \$mapping.functions[\$func] = [ordered]@{
        test_file = \$func + '.Tests.ps1'
        has_test = \$testExists
    }
}

# Calculate coverage
\$covered = \$mapping.functions.Values | Where-Object { \$_.has_test } | Measure-Object | Select-Object -ExpandProperty Count
\$mapping.coverage = [ordered]@{
    covered = \$covered
    total = \$publicFunctions.Count
    percentage = if (\$publicFunctions.Count -gt 0) { [math]::Round((\$covered / \$publicFunctions.Count) * 100, 2) } else { 0 }
}

# Output JSON
\$mapping | ConvertTo-Json -Depth 10 | Out-File -FilePath './Tests/Mapping/function-test-map.json' -Force

# Report
Write-Host "Mapping generated: \$covered / \$(\$publicFunctions.Count) public functions have tests (\$(\$mapping.coverage.percentage)%)"
if (\$covered -lt \$publicFunctions.Count) {
    Write-Host "Functions missing tests:"
    \$mapping.functions.GetEnumerator() | Where-Object { -not \$_.Value.has_test } | ForEach-Object {
        Write-Host "  - \$(\$_.Key)"
    }
}

exit 0
```
